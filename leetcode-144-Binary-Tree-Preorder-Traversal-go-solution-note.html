题目 leetcode 144 Binary Tree Preorder Traversal

对一个二叉树做pre order 遍历

解题思路分析

直接递归做

java题解代码

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        visit(root, list);
        return list;
    }
    
    public void visit(TreeNode root, List<Integer> list) {
        if(null == root) return;
        list.add(root.val);
        visit(root.left, list);
        visit(root.right, list);
    }
}

java非递归做法(Stack)

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        if(null == root) return list;
        
        Stack<TreeNode> s = new Stack<>();    
        TreeNode node = root;
        s.push(node);
        while(s.size() > 0) {
            node = s.pop();
            list.add(node.val);
            
            if(null != node.right) s.push(node.right);
            if(null != node.left) s.push(node.left);
        }
        
        return list;
    }
    
}

go题解代码

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func preorderTraversal(root *TreeNode) []int {
    list := make([]int, 0)
    dfs(root, &list)
    return list
}

func dfs(node *TreeNode, list *[]int) {
    if node == nil { return }
    (*list) = append(*list, node.Val)
    dfs(node.Left, list)
    dfs(node.Right, list)
}
