题目 – leetcode 426 Convert Binary Search Tree to Sorted Doubly Linked List

输入一个二叉搜索树， 转换成一个双向链表

解题思路分析

  1. 基本思路是in order 遍历， 这个题目前两天写过， 具体在http://www.noteanddata.com/leetcode-426-Convert-Binary-Search-Tree-to-Sorted-Doubly-Linked-List-java-solution-note.html
  2. 今天再写一次，感觉代码还可以略简洁一点, tail其实不需要， prev就是tail

java题解代码

/*
// Definition for a Node.
class Node {
    public int val;
    public Node left;
    public Node right;

    public Node() {}

    public Node(int _val,Node _left,Node _right) {
        val = _val;
        left = _left;
        right = _right;
    }
};
*/
class Solution {
    private Node head, prev;
    public Node treeToDoublyList(Node root) {
        if(null == root) return null;
        head = prev = null;
        dfs(root);
        prev.right = head;
        head.left = prev;
        return head;
    }
    private void dfs(Node node) {
        if(null == node) return;
        dfs(node.left);
        Node cur = new Node(node.val, null, null);
        if(null == head) {
            head = cur;
        }
        if(null != prev) {
            prev.right = cur;
            cur.left = prev;
        }
        prev = cur;
        dfs(node.right);
    }
}

时间复杂度

O(N)

空间复杂度

O(lgN)

java题解代码2

前面用了全局的变量, 其中head是必须的， prev其实可以通过参数传入

/*
// Definition for a Node.
class Node {
    public int val;
    public Node left;
    public Node right;

    public Node() {}

    public Node(int _val,Node _left,Node _right) {
        val = _val;
        left = _left;
        right = _right;
    }
};
*/
class Solution {
    private Node head;
    public Node treeToDoublyList(Node root) {
        if(null == root) return null;
        Node last = dfs(root, null);
        
        last.right = head;
        head.left = last;
        return head;
    }
    private Node dfs(Node node, Node prev) {
        if(null == node) return null;
        
        Node leftRet = dfs(node.left, prev);    
        if(null != leftRet) {
            prev = leftRet;
        }
        
        Node cur = new Node(node.val, null, null);
        if(null != prev) {
            cur.left = prev;
            prev.right = cur;            
        }
        
        if(null == head) {
            head = cur;
        }
        
        Node rightRet = dfs(node.right, cur);
        return (null != rightRet) ? rightRet : cur;
    }
}

前面对于null的处理，可以更加简洁一点， 如果最开始null判断的时候， 直接return prev的话，中间一些地方可以更加简洁一点

/*
// Definition for a Node.
class Node {
    public int val;
    public Node left;
    public Node right;

    public Node() {}

    public Node(int _val,Node _left,Node _right) {
        val = _val;
        left = _left;
        right = _right;
    }
};
*/
class Solution {
    private Node head;
    public Node treeToDoublyList(Node root) {
        if(null == root) return null;
        Node last = dfs(root, null);
        
        last.right = head;
        head.left = last;
        return head;
    }
    private Node dfs(Node node, Node prev) {
        if(null == node) return prev;
        
        prev = dfs(node.left, prev);    
        
        Node cur = new Node(node.val, null, null);
        if(null != prev) {
            cur.left = prev;
            prev.right = cur;            
        }
        
        if(null == head) {
            head = cur;
        }
        
        Node rightRet = dfs(node.right, cur);
        return rightRet;
    }
}
