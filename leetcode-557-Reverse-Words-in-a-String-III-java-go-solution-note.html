题目 leetcode 557 Reverse Words in a String III

对字符串里面的每个单词做reverse, 单词之间用空格隔开

解题思路分析

  1. 思路非常直接， 用空格判断单词，然后对单词进行reverse就好

java题解代码

class Solution {
    public String reverseWords(String s) {
        if(null == s) return s;
        char[] arr = s.toCharArray();
        int begin = 0;
        for(int i = 0; i < arr.length; ++i) {
            if(arr[i] == ' ') {
                reverse(arr, begin, i-1);
                begin = i+1;
            }
        }
                    
        if(begin < arr.length) {
            reverse(arr, begin, arr.length-1);    
        }
        
        return new String(arr);
    }
    
    public void reverse(char[] arr, int p, int q) {
        while(p < q) {
            char t = arr[p];
            arr[p] = arr[q];
            arr[q] = t;

            p++;
            q--;
        }        
    }
}

go题解代码

func reverseWords(s string) string {
    arr := []byte(s)
    begin := 0
    for i := 0; i < len(s); i++ {
        if s[i] == ' ' {
            reverse(arr, begin, i-1)
            begin = i+1
        }
    }
    if begin < len(s) {
        reverse(arr, begin, len(s)-1)
    }
    return string(arr)
}

func reverse(arr []byte, begin int, end int) {
    for ; begin < end; {
        temp := arr[begin]
        arr[begin] = arr[end]
        arr[end] = temp
        begin++
        end--
    }
}
