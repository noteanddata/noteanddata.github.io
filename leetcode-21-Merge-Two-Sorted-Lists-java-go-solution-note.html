题目 leetcode 21 Merge Two Sorted Lists

输入两个sorted linked list, 要求合并成一个同样sorted linked list

解题思路分析

  1. 两个指针分别移动就好， 迭代的方式更好一点， 当然递归做的话代码会更简洁一点但是空间复杂度会高

java题解代码

这个代码以前写的， 现在看起来其实问题很多， 主要是每次new一个其实没有必要， 应该直接在原来的指针上操作。
如果用一个dummy节点的话，可以写的比较简洁一点

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        ListNode resultHead = null;
        ListNode node = null;
        while(l1 != null && l2 != null) {
            int nextVal = l1.val;
            if(l1.val <= l2.val) {
                l1 = l1.next;
            }
            else {
                nextVal = l2.val;
                l2 = l2.next;
            }

            if(node == null) {
                resultHead = node = new ListNode(nextVal);
            }
            else {
                node = node.next = new ListNode(nextVal);
            }
        }
        
        while(l1 != null) {
            if(node == null) {
                resultHead = node = new ListNode(l1.val);
            }
            else {
                node = node.next = new ListNode(l1.val);
            }
            l1 = l1.next;
        }

        while(l2 != null) {
            if(node == null) {
                resultHead = node = new ListNode(l2.val);
            }
            else {
                node = node.next = new ListNode(l2.val);
            }
            l2 = l2.next;
        }
        

        return resultHead;
    }}

go语言题解代码

go语言里面一些不方便的地方
- 没有三元操作符号
- if语句不能省略括号
- 赋值不能多个变量串起来赋值


/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func mergeTwoLists(l1 *ListNode, l2 *ListNode) *ListNode {
    if l1 == nil {
        return l2
    }
    if l2 == nil {
        return l1
    }
    var nhead *ListNode
    var cur *ListNode
    for ; l1 != nil && l2 != nil; {
        if l1.Val < l2.Val {
            if cur == nil {
                cur = l1
                nhead = l1
            } else {
                cur.Next = l1
                cur = cur.Next
            }
            l1 = l1.Next
        } else {
            if cur == nil {
                cur = l2
                nhead = l2
            } else {
                cur.Next = l2
                cur = cur.Next
            }
            l2 = l2.Next
        }
    }
    
    if l1 != nil {
        cur.Next = l1
    } else {
        cur.Next = l2
    }
    return nhead
}
