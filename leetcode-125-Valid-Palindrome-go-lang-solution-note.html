题目 leetcode 125 Valid Palindrome

判断一个字符串是否回文， 但是只考虑其中的字母和数字

解题思路分析

头尾两个指针， 然后跳过非字母和数字的字符
http://www.noteanddata.com/leetcode-125-Valid-Palindrome-facebook-interview-problem-java-solution-note.html

go语言题解代码

import "unicode"
func isPalindrome(s string) bool {
    for left, right := 0, len(s)-1; left < right; {
        for ; left < right && (!unicode.IsLetter(int32(s[left])) && !unicode.IsDigit(int32(s[left]))); left++ {
            
        }
        for ; left < right && (!unicode.IsLetter(int32(s[right])) && !unicode.IsDigit(int32(s[right]))); right-- {
            
        }
        if left >= right {
            break
        }
        if unicode.ToLower(int32(s[left])) != unicode.ToLower(int32(s[right])) {
            //fmt.Printf("%d, %c, %d, %c \n", left, s[left], right,  s[right])
            return false
        }
        left++
        right--
    }
    return true
}
