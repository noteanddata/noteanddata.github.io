题目 leetcode 896 Monotonic Array

判断一个数组是否是单调的

解题思路分析

第一次遇到有大小的时候定好当前的方向

java题解代码

class Solution {
    public boolean isMonotonic(int[] A) {
        if(A.length <= 2) {
            return true;
        }
        Boolean inc = null;
        for(int i = 0; i < A.length-1; ++i) {
            if(A[i] < A[i+1]) {
                if(null == inc) {
                    inc = true;
                }
                else if(inc) {
                    continue;
                }
                else {
                    return false;
                }
            }
            else if(A[i] > A[i+1]) {
                if(null == inc) {
                    inc = false;
                }
                else if(!inc) {
                    continue;
                }
                else {
                    return false;
                }
            }
        }
        return true;
    }
}

go语言题解代码

func isMonotonic(A []int) bool {
    dir := 0
    for i := 1; i < len(A); i++ {
        if A[i] > A[i-1] {
            if dir == 0 {
                dir = 1 // increase
            } else if dir == 2 {
                return false
            }
        } else if A[i] < A[i-1] {
            if dir == 0 {
                dir = 2
            } else if dir == 1 {
                return false
            }
        }
    }
    return true
}

go语言错误代码

中间会有一些元素是大小相等的，所以直接判断连续的三个是不对的
Input: [11,11,9,4,3,3,3,1,-1,-1,3,3,3,5,5,5]
Output: true
Expected: false

func isMonotonic(A []int) bool {
    for i := 2; i < len(A); i++ {
        if (A[i] > A[i-1] && A[i-1] < A[i-2]) || (A[i] < A[i-1] && A[i-1] > A[i-2]) {
            return false
        }
    }
    return true
}
