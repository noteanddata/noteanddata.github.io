题目 leetcode 415 Add Strings

输入两个字符串，里面都是数字， 对两个字符串进行相加操作

解题思路分析

直接从后面加到前面就好， 最后记得carry要加上去

java题解代码

public String addStrings(String num1, String num2) {
    StringBuilder sb = new StringBuilder();
    int carry = 0;
    for(int i = num1.length()-1, j = num2.length()-1; i >= 0 || j >= 0; i--, j--) {
        int a = i >= 0 ? (num1.charAt(i) - '0') : 0;
        int b = j >= 0 ? (num2.charAt(j) - '0') : 0;
        int v = a + b + carry;
        char ch = (char)('0' + v%10);
        sb.append(ch);
        carry = v/10;
    }
    if(carry > 0) {
        char ch = (char)('0' + carry);
        sb.append(ch);
    }
    return sb.reverse().toString();
}

//import "strings"
func addStrings(num1 string, num2 string) string {
    var s string
    carry := byte(0)
    for i,j := len(num1)-1, len(num2)-1; i >= 0 || j >= 0; {
        a := byte(0)
        if i >= 0 {
            a = num1[i] - '0'
        }
        
        b := byte(0)
        if j >= 0 {
            b = num2[j] - '0'
        }
        var ret = a+b+carry
        s = string('0' + ret%10) + s
        //sb.WriteByte(ret%10)
        carry = ret/10
        
        i--;
        j--;
    }
    if carry > 0 {
        //sb.WriteByte(carry)
        s = string('0' + carry) + s
    }
    return s
}
