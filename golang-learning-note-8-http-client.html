最基础的代码

下面就是最基础的代码

package main 
import (
  "net/http"
  "io/ioutil"
  "fmt"
)
func main() {
  run()
}

func run() {
  resp, _ := http.Get("http://example.com")
  defer resp.Body.Close()
  
  body, _ := ioutil.ReadAll(resp.Body)
  fmt.Println(string(body))
}
