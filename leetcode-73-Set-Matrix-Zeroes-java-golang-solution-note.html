题目 leetcode 73 Set Matrix Zeroes

输入一个矩阵， 如果某一行有0， 那么这行全部改成0， 如果某一列有0， 那么这列全部改成0

解题思路分析

  1. 需要先用一个list把有0的行保存起来， 然后再用一个list把有0的列保存起来， 然后做修改。
  2. 不能直接修改， 因为会覆盖掉结果
  3. 好像O(1)空间的做法是利用矩阵本身的第一行和第一列做修改 (具体还没有仔细看)

java题解代码

class Solution {
    public void setZeroes(int[][] matrix) {
        if(null == matrix || 0 == matrix.length || 0 == matrix[0].length) return ;
        
        List<Integer> zerocolList = new ArrayList<>();
        List<Integer> zerorowList = new ArrayList<>();
        for(int i = 0; i < matrix.length; ++i) {            
            for(int j = 0; j < matrix[i].length; ++j) {
                if(matrix[i][j] == 0) {
                    zerocolList.add(j);
                    zerorowList.add(i);
                }
            }
        }
        
        for(int row: zerorowList) {
            for(int j = 0; j < matrix[0].length; ++j) {
                matrix[row][j] = 0;
            }
        }
        for(int i = 0; i < matrix.length; ++i) {
            for(int col: zerocolList) {
                matrix[i][col] = 0;
            }
        }
        
    }
}

go题解代码

func setZeroes(matrix [][]int)  {
    rows := make([]int, 0)
    cols := make([]int, 0)
    for i := 0; i < len(matrix); i++ {
        for j := 0; j < len(matrix[i]); j++ {
            if matrix[i][j] == 0 {
                rows = append(rows, i)
                cols = append(cols, j)
            }
        }
    }
    
    for _, row := range rows {
        for j := 0; j < len(matrix[0]); j++ {
            matrix[row][j] = 0
        }
    }
    
    for _, col := range cols {
        for i := 0; i < len(matrix); i++ {
            matrix[i][col] = 0
        }
    }
}
