题目 leetcode 2 Add Two Numbers

go语言题解代码

不过整体感觉这个代码写的非常繁琐, 前面的java写的也比较繁琐，下次看看写的简洁一点

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    var head *ListNode
    var cur *ListNode
    carry := 0
    for ; l1 != nil && l2 != nil; l1, l2 = l1.Next, l2.Next {
        v := l1.Val + l2.Val + carry
        carry = v/10
        v = v%10

        if head == nil {
            head = &ListNode{v, nil}
            cur = head
        } else {
            cur.Next = &ListNode{v, nil}
            cur = cur.Next
        }
        
    }    
    if l1 != nil || l2 != nil {
        if l2 != nil {
            l1 = l2
        }
        for ; l1 != nil; l1 = l1.Next {
            v := l1.Val + carry
            carry = v/10
            v = v%10

            if head == nil {
                head = &ListNode{v, nil}
                cur = head
            } else {
                cur.Next = &ListNode{v, nil}
                cur = cur.Next
            }
        }
    }
    if carry > 0 {
        cur.Next = &ListNode{carry, nil}
    }
    return head
}
