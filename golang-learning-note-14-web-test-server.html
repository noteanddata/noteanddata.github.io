go语言实现简单的测试用的http服务器

01 September 2019

写一个简单的http服务器，可以打印收到的http request

代码如下， 主要功能很简单，就是打印收到的http请求的header和body，
主要目的是在写gourl的过程中方便测试

package main
import "net/http"
import "log"
import "io"
import "io/ioutil"

func main() {
  h1 := func(w http.ResponseWriter, r *http.Request) {
    log.Println("URL=", r.URL)
    log.Println("Method=", r.Method)
    log.Println("Proto=", r.Proto)
    for k, v := range r.Header {
      log.Println(k, "=", v)
    }
    body, _ := ioutil.ReadAll(r.Body)
    log.Printf("%s \n", body)
    
    io.WriteString(w, "Hello from a HandleFunc #1!\n")
  }
  
  http.HandleFunc("/", h1)

	log.Fatal(http.ListenAndServe(":8080", nil))
}
