题目 leetcode 9 Palindrome Number

判断一个数是否是回文数

解题思路分析

可以转换成字符串， 也可以直接数字判断， 注意尽量不要有溢出
http://www.noteanddata.com/leetcode-9-Palindrome-Number-java-solution-note.html

java题解代码

public  boolean isPalindrome(int x) {
   if(x < 0 || (x != 0 && x%10 == 0)) return false;
   int mirror = 0;
   while(x > mirror) {
       mirror = mirror * 10 + x%10;
       x /= 10;
   }
   return mirror == x || mirror/10 == x;
}

go语言题解代码1

func isPalindrome(x int) bool {
    if x < 0 || (x != 0 && x%10 == 0) {
        return false
    }
    mirror := 0
    for ; mirror < x; {
        r := x%10
        mirror = mirror * 10 + r
        x = x/10
    }
    return mirror == x || mirror/10 == x
}

go语言题解代码2

用strconv.Itoa把数字转换成字符串

import "strconv"
func isPalindrome(x int) bool {
    s := strconv.Itoa(x)
    for left, right := 0, len(s)-1; left < right; {
        if s[left] != s[right] {
            return false
        }
        left++
        right--
    }
    return true
}
