题目 leetcode 111 Minimum Depth of Binary Tree

返回一个二叉树的最低高度， 最低高度是指从根节点到叶节点的高度

解题思路分析

递归调用， 但是这个题目稍微要注意一点， 需要的是返回到叶节点的高度， 如果没有到叶节点的高度是不算的

java错误代码

一开始想直接递归了， 但是这样的问题是这样返回的高度不一定会是到叶节点的

public int minDepth(TreeNode root) {
    if(root == null) return 0;
    return 1+Math.min(minDepth(root.left), minDepth(root.right));
}

java题解代码1

public int minDepth(TreeNode root) {
    if(root == null) return 0;
    if(root.left == null && root.right == null) return 1;
    if(root.left == null) return 1+minDepth(root.right);
    if(root.right == null) return 1+minDepth(root.left);
    return 1+Math.min(minDepth(root.left), minDepth(root.right));
}

java题解代码2

level order BFS

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minDepth(TreeNode root) {
        if(root == null) return 0;
        Queue<TreeNode> queue = new LinkedList<>();
        int height = 1;
        queue.add(root);
        Queue<TreeNode> next = new LinkedList<>();
        while(true) {
            while(queue.size() > 0) {
                TreeNode node = queue.poll();
                if(null == node.left && null == node.right) {
                    return height;
                }
                if(null != node.left) {
                    next.add(node.left);
                }
                if(null != node.right) {
                    next.add(node.right);
                }
            }
            queue = next;
            next = new LinkedList<>();
            height++;
        }        
    }
}

go语言题解代码1

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func minDepth(root *TreeNode) int {
    if nil == root {
        return 0
    }    
    if root.Left == nil && root.Right == nil {
        return 1
    }
    leftDepth := minDepth(root.Left)
    rightDepth := minDepth(root.Right)
    if root.Left == nil {
        return 1 + rightDepth
    } else if root.Right == nil {
        return 1 + leftDepth
    } else {
        if leftDepth < rightDepth {
            return 1 + leftDepth
        } else {
            return 1+rightDepth
        }
    }
}

go语言题解代码2

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func minDepth(root *TreeNode) int {
    if nil == root {
        return 0
    }    
    if root.Left == nil && root.Right == nil {
        return 1
    }
    leftDepth := minDepth(root.Left)
    rightDepth := minDepth(root.Right)
    if root.Left == nil  || (root.Right != nil && leftDepth > rightDepth) {
        return 1 + rightDepth
    } else {
        return 1 + leftDepth
    } 
}
