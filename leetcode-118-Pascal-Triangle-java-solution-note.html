leetcode 118 Pascal’s Triangle

输出一个Pascal三角形

java代码实现

这个题目直接按照逻辑来做就好， 代码尽量简洁一点， 下面是我目前可以写到到版本

public List<List<Integer>> generate(int numRows) {
    List<List<Integer>> allList = new ArrayList<>();
    for(int i = 0; i < numRows; ++i) {
        List<Integer> list = new ArrayList<>();
        if(i == 0) {
            list.add(1);
        }
        else {
            list.add(1);
            List<Integer> prevList = allList.get(i-1);
            for(int j = 1; j < i; ++j) {
                list.add(prevList.get(j-1) + prevList.get(j));
            }
            list.add(1);
        }
        allList.add(list);
    }
    return allList;
}

public List<List<Integer>> generate(int numRows) {
    List<List<Integer>> allList = new ArrayList<>();
    for(int i = 0; i < numRows; ++i) {
        List<Integer> list = new ArrayList<>();
        if(i < 2) {                
            for(int j = 0; j <= i; ++j) {
                list.add(1);
            }
            allList.add(list);
        }
        else {
            list.add(1);
            List<Integer> prev = allList.get(i-1);
            for(int j = 1; j < i; ++j) {
                list.add(prev.get(j-1) + prev.get(j));
            }
            list.add(1);
            allList.add(list);
        }
    }
    return allList;
}
