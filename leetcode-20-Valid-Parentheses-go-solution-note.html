题目 leetcode 20 Valid Parentheses

一个字符串里面有’(’, ‘)’, ‘{’, ‘}’, ‘[’ and ’]’, 判断这个字符串是否是合法的匹配的括号字符串

解题思路分析

直接用一个stack来做, 更多分析参考 http://www.noteanddata.com/leetcode-20-Valid-Parentheses-java-solution-note.html

go语言题解代码

func isValid(s string) bool {
    stack := make([]int32, 0)
    for _, v := range s {
        if v == '(' || v == '[' || v == '{' {
            stack = append(stack, v)
        } else {
            if len(stack) == 0 {
                return false
            } 
            if v == ')' {
                if stack[len(stack)-1] != '(' {
                    return false
                }
            } else if v == ']' {
                if stack[len(stack)-1] != '[' {
                    return false
                }
            } else {
                if stack[len(stack)-1] != '{' {
                    return false
                }                
            }
            
            stack = stack[0: len(stack)-1]
        }
    }
    return len(stack) == 0
}
